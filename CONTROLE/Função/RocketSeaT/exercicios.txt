Crie um botão que ao ser clicado cria um novo elemento em tela com a forma de um quadrado
vermelho com 100px de altura e largura. Sempre que o botão for clicado um novo quadrado deve
aparecer na tela.



<div id="app">
        <button class="botao">Clique Aqui</button>
        <div class="box"></div>
    </div>

    <script>
        var btnElement = document.querySelector('button.botao');
       
            btnElement.onclick = function () {
            
            var boxElement = document.createElement('div');
            boxElement.style.width = 200;
            boxElement.style.height = 200;
            boxElement.style.backgroundColor = '#f00';
            boxElement.style.margin = 10;
            
            document.querySelector('.box').appendChild(boxElement);
            
        }



    </script>

Utilizando o resultado do primeiro desafio, toda vez que o usuário passar o mouse por cima de
algum quadrado troque sua cor para uma cor aleatória gerada pela função abaixo:


<div id="app">
        <button class="botao">Clique Aqui</button>
        <div class="box"></div>
    </div>

    <script>
        var btnElement = document.querySelector('button.botao');

        btnElement.onclick = function () {

            var boxElement = document.createElement('div');
            boxElement.style.width = 200;
            boxElement.style.height = 200;
            boxElement.style.backgroundColor = '#f00';
            boxElement.style.margin = 10;
            
            boxElement.addEventListener('mouseover',()=>{
                var newColor = getRandomColor();
                boxElement.style.backgroundColor=newColor;
            })

            document.querySelector('.box').appendChild(boxElement);

        }
        

        function getRandomColor() {
            var letters = "0123456789ABCDEF";
            var color = "#";
            for (var i = 0; i < 6; i++) {
                color += letters[Math.floor(Math.random() * 16)];
            }
            return color;
        }
        



    </script>

● Diego

        <ul></ul>
    </div>

    <script>
       var nomes=["Diego", "Gabriel", "Lucas"];
       
       var listaElement = document.querySelector('ul');

       for(nomes of nomes){
           var liElement = document.createElement('li');
